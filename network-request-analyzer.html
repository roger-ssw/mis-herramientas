Focused on what matters:

This tool ONLY captures requests to ad platforms (Meta, Google Ads, TikTok, etc.), tracking/attribution tools (Hyros, AnyTrack, Triple Whale), and GTM. It also compiles a full DataLayer event timeline with e-commerce data validation. Run it, navigate the site, then get a clean report of exactly what's firing and what data it's sending.

  • Meta / Facebook (Pixel + CAPI signals)
  • Google Ads (conversions, remarketing)
  • Google Analytics (GA4 collect)
  • TikTok Events API
  • Pinterest Conversions
  • Snapchat Pixel
  • LinkedIn Insight
  • Twitter / X Pixel
  • Reddit Pixel

  • Custom event capture
  • Live monitoring (new pushes)

(function() {
    console.clear();
    console.log('%c AD & TRACKING NETWORK ANALYZER v1.0', 'background: #0891b2; color: white; font-size: 16px; padding: 10px; border-radius: 5px;');
    console.log('%c Ad platforms, GTM, DataLayer & attribution tracking', 'color: #a0a0a0; font-style: italic;');
    console.log('%c' + '='.repeat(55), 'color: #0891b2;');

    var startTime = Date.now();
    var adRequests = [];
    var dataLayerTimeline = [];
    var fbqEvents = [];
    var gtagEvents = [];

    // ─── AD & TRACKING PLATFORMS ONLY ────────────────────────
    var adPlatforms = [
        { name: 'Meta Pixel', patterns: ['facebook.com/tr', 'facebook.net/en_US/fbevents', 'connect.facebook.net'] },
        { name: 'Google Ads', patterns: ['googleads.g.doubleclick.net', 'googleadservices.com/pagead', 'www.googleadservices.com'] },
        { name: 'Google Analytics', patterns: ['google-analytics.com', 'analytics.google.com', '/g/collect', '/j/collect'] },
        { name: 'GTM', patterns: ['googletagmanager.com/gtm.js', 'googletagmanager.com/gtag'] },
        { name: 'Google DCM', patterns: ['doubleclick.net/activity', 'fls.doubleclick.net'] },
        { name: 'TikTok', patterns: ['analytics.tiktok.com', 'business-api.tiktok.com'] },
        { name: 'Pinterest', patterns: ['ct.pinterest.com', 'pinterest.com/v3'] },
        { name: 'Snapchat', patterns: ['tr.snapchat.com', 'sc-static.net/scevent'] },
        { name: 'LinkedIn', patterns: ['linkedin.com/px', 'linkedin.com/collect', 'snap.licdn.com', 'linkedin.com/li/'] },
        { name: 'Twitter / X', patterns: ['ads-twitter.com', 'static.ads-twitter.com', 'analytics.twitter.com', 't.co/i/adsct'] },
        { name: 'Reddit', patterns: ['alb.reddit.com/snoo', 'events.reddit.com'] },
        { name: 'Hyros', patterns: ['hyros.com', 't.hyros.com', 'api.hyros.com'] },
        { name: 'AnyTrack', patterns: ['anytrack.io', 'cdn.anytrack.io'] },
        { name: 'Triple Whale', patterns: ['triplewhale.com', 'api.triplewhale.com'] },
        { name: 'SegMetrics', patterns: ['segmetrics.io'] },
        { name: 'Wicked Reports', patterns: ['wickedreports.com'] },
        { name: 'Segment', patterns: ['api.segment.io', 'cdn.segment.com'] },
        { name: 'Klaviyo', patterns: ['a.]klaviyo.com/api/track', 'klaviyo.com/onsite'] },
        { name: 'ActiveCampaign', patterns: ['activehosted.com/proc'] }
    ];

    function matchPlatform(url) {
        if (!url) return null;
        for (var i = 0; i < adPlatforms.length; i++) {
            for (var j = 0; j < adPlatforms[i].patterns.length; j++) {
                if (url.indexOf(adPlatforms[i].patterns[j]) !== -1) return adPlatforms[i].name;
            }
        }
        return null;
    }

    // ─── PAYLOAD PARSER ──────────────────────────────────────
    function parsePayload(url, body) {
        var data = {};

        // Parse URL params
        try {
            var urlObj = new URL(url);
            urlObj.searchParams.forEach(function(val, key) {
                data[key] = val;
            });
        } catch(e) {
            // Try to extract query string manually
            var qIdx = url.indexOf('?');
            if (qIdx !== -1) {
                url.substring(qIdx + 1).split('&').forEach(function(pair) {
                    var parts = pair.split('=');
                    if (parts[0]) data[decodeURIComponent(parts[0])] = decodeURIComponent(parts[1] || '');
                });
            }
        }

        // Parse body
        if (body && typeof body === 'string') {
            try {
                var parsed = JSON.parse(body);
                Object.keys(parsed).forEach(function(k) { data['body.' + k] = parsed[k]; });
            } catch(e) {
                // URL-encoded body
                body.split('&').forEach(function(pair) {
                    var parts = pair.split('=');
                    if (parts[0]) {
                        try { data['body.' + decodeURIComponent(parts[0])] = decodeURIComponent(parts[1] || ''); }
                        catch(e2) { data['body.' + parts[0]] = parts[1] || ''; }
                    }
                });
            }
        }

        return data;
    }

    function extractEventInfo(platform, payload) {
        var info = {};

        // Meta Pixel
        if (platform === 'Meta Pixel') {
            if (payload.ev) info.event = payload.ev;
            if (payload.id) info.pixelId = payload.id;
            if (payload.cd) {
                try { info.customData = JSON.parse(payload.cd); } catch(e) { info.customData = payload.cd; }
            }
            if (payload.ud) info.userData = 'present (hashed)';
            if (payload['body.event_id'] || payload.eid) info.eventId = payload['body.event_id'] || payload.eid;
        }

        // Google Analytics
        if (platform === 'Google Analytics') {
            if (payload.en) info.event = payload.en;
            if (payload.tid) info.measurementId = payload.tid;
            if (payload.ep) info.eventParams = payload.ep;
            if (payload.v) info.version = payload.v;
        }

        // Google Ads
        if (platform === 'Google Ads') {
            if (payload.label) info.conversionLabel = payload.label;
            if (payload.value) info.value = payload.value;
            if (payload.currency) info.currency = payload.currency;
        }

        // TikTok
        if (platform === 'TikTok') {
            if (payload.event) info.event = payload.event;
            if (payload['body.event']) info.event = payload['body.event'];
        }

        // Hyros
        if (platform === 'Hyros') {
            Object.keys(payload).forEach(function(k) {
                if (k.indexOf('body.') === 0) info[k.replace('body.', '')] = payload[k];
            });
        }

        return info;
    }

    function addAdRequest(entry) {
        entry.timestamp = new Date().toISOString();
        entry.elapsed = ((Date.now() - startTime) / 1000).toFixed(1) + 's';
        entry.payload = parsePayload(entry.url, entry.body);
        entry.eventInfo = extractEventInfo(entry.platform, entry.payload);

        adRequests.push(entry);

        var eventStr = entry.eventInfo.event ? ' [' + entry.eventInfo.event + ']' : '';
        var idStr = entry.eventInfo.pixelId ? ' (ID: ' + entry.eventInfo.pixelId + ')' : '';
        console.log('  %c' + entry.platform + '%c' + eventStr + idStr + ' (' + entry.type + ', ' + entry.elapsed + ')',
            'color: #FF8C42; font-weight: bold;', 'color: #a0a0a0;');
    }

    // ─── 1. DATALAYER COMPILATION ────────────────────────────
    console.log('\n%c DATALAYER COMPILATION', 'background: #7c3aed; color: white; padding: 5px; border-radius: 3px;');

    var ga4EcomEvents = ['view_item', 'view_item_list', 'select_item', 'add_to_cart', 'remove_from_cart',
        'view_cart', 'begin_checkout', 'add_shipping_info', 'add_payment_info', 'purchase', 'refund'];

    function analyzeDataLayerEntry(item, index) {
        var entry = {
            index: index,
            timestamp: new Date().toISOString(),
            event: item.event || null,
            raw: item
        };

        // Consent events
        if (item[0] === 'consent') {
            entry.type = 'consent';
            entry.consentAction = item[1];
            entry.consentConfig = item[2];
        }
        // GA4 e-commerce
        else if (item.event && ga4EcomEvents.indexOf(item.event) !== -1) {
            entry.type = 'ecommerce';
            var ecom = item.ecommerce || item;
            entry.ecommerce = {
                hasItems: !!(ecom.items || (item.ecommerce && item.ecommerce.items)),
                hasValue: !!(ecom.value || (item.ecommerce && item.ecommerce.value)),
                hasCurrency: !!(ecom.currency || (item.ecommerce && item.ecommerce.currency)),
                hasTransactionId: !!(ecom.transaction_id || (item.ecommerce && item.ecommerce.transaction_id))
            };
            entry.ecommerce.missing = [];
            if (!entry.ecommerce.hasItems) entry.ecommerce.missing.push('items');
            if (!entry.ecommerce.hasValue) entry.ecommerce.missing.push('value');
            if (!entry.ecommerce.hasCurrency) entry.ecommerce.missing.push('currency');
            if (item.event === 'purchase' && !entry.ecommerce.hasTransactionId) entry.ecommerce.missing.push('transaction_id');

            // Count items if present
            var items = ecom.items || (item.ecommerce && item.ecommerce.items);
            if (items && Array.isArray(items)) {
                entry.ecommerce.itemCount = items.length;
                entry.ecommerce.totalValue = ecom.value || (item.ecommerce && item.ecommerce.value) || 'not set';
            }
        }
        // GTM events
        else if (item.event && item.event.indexOf('gtm.') === 0) {
            entry.type = 'gtm';
        }
        // Custom events
        else if (item.event) {
            entry.type = 'custom';
        }
        // Data push (no event)
        else {
            entry.type = 'data';
        }

        return entry;
    }

    if (window.dataLayer) {
        console.log('  Existing dataLayer entries: ' + window.dataLayer.length);

        // Compile existing entries
        window.dataLayer.forEach(function(item, i) {
            var entry = analyzeDataLayerEntry(item, i);
            dataLayerTimeline.push(entry);

            if (entry.type === 'ecommerce') {
                var status = entry.ecommerce.missing.length ? '%c MISSING: ' + entry.ecommerce.missing.join(', ') : '%c OK';
                var color = entry.ecommerce.missing.length ? 'color: #ef4444;' : 'color: #10b981;';
                console.log('  [ecom] ' + entry.event + ' — items: ' + (entry.ecommerce.itemCount || '?') + ', value: ' + entry.ecommerce.totalValue + ' ' + status, color);
            } else if (entry.type === 'consent') {
                console.log('  [consent] ' + entry.consentAction + ': ' + JSON.stringify(entry.consentConfig));
            }
        });

        // Monitor new pushes
        var origPush = window.dataLayer.push;
        window.dataLayer.push = function() {
            var result = origPush.apply(this, arguments);

            for (var i = 0; i < arguments.length; i++) {
                var item = arguments[i];
                var entry = analyzeDataLayerEntry(item, window.dataLayer.length - 1);
                entry.live = true;
                dataLayerTimeline.push(entry);

                if (entry.event) {
                    var prefix = entry.type === 'ecommerce' ? '[ecom]' : entry.type === 'consent' ? '[consent]' : '[event]';
                    console.log('  %c[LIVE] ' + prefix + ' ' + entry.event, 'color: #FF8C42; font-weight: bold;');

                    if (entry.type === 'ecommerce' && entry.ecommerce.missing.length) {
                        console.log('    %cMISSING: ' + entry.ecommerce.missing.join(', '), 'color: #ef4444;');
                    }
                }
            }

            return result;
        };
        console.log('  dataLayer.push() interceptor active — live events will appear');
    } else {
        console.log('  No dataLayer found');
    }

    // ─── 2. FBQ INTERCEPTOR ──────────────────────────────────
    console.log('\n%c FBQ EVENT INTERCEPTOR', 'background: #1877f2; color: white; padding: 5px; border-radius: 3px;');

    if (window.fbq) {
        var origFbq = window.fbq;
        window.fbq = function() {
            var args = Array.from(arguments);
            var entry = {
                timestamp: new Date().toISOString(),
                elapsed: ((Date.now() - startTime) / 1000).toFixed(1) + 's',
                action: args[0],
                event: args[1],
                params: args[2] || {},
                hasEventId: !!(args[2] && args[2].event_id),
                hasExternalId: !!(args[2] && args[2].external_id),
                hasUserData: !!(args[2] && (args[2].em || args[2].ph || args[2].fn))
            };
            fbqEvents.push(entry);

            if (args[0] === 'track' || args[0] === 'trackCustom') {
                var capiFlag = entry.hasEventId ? ' %c[CAPI]' : '';
                console.log('  %c[fbq] ' + args[0] + ': ' + args[1] + capiFlag,
                    'color: #1877f2; font-weight: bold;',
                    entry.hasEventId ? 'color: #10b981; font-weight: bold;' : '');
                if (args[2]) {
                    var params = Object.keys(args[2]).filter(function(k) { return ['value', 'currency', 'content_type', 'content_ids', 'event_id', 'external_id'].indexOf(k) !== -1; });
                    if (params.length) {
                        var details = params.map(function(k) { return k + ': ' + args[2][k]; }).join(', ');
                        console.log('    ' + details);
                    }
                }
            }

            return origFbq.apply(this, arguments);
        };
        console.log('  fbq interceptor active');
    } else {
        console.log('  fbq not found');
    }

    // ─── 3. GTAG INTERCEPTOR ─────────────────────────────────
    console.log('\n%c GTAG EVENT INTERCEPTOR', 'background: #ea4335; color: white; padding: 5px; border-radius: 3px;');

    if (window.gtag) {
        var origGtag = window.gtag;
        window.gtag = function() {
            var args = Array.from(arguments);
            var entry = {
                timestamp: new Date().toISOString(),
                elapsed: ((Date.now() - startTime) / 1000).toFixed(1) + 's',
                command: args[0],
                eventOrId: args[1],
                params: args[2] || {}
            };
            gtagEvents.push(entry);

            if (args[0] === 'event') {
                console.log('  %c[gtag] event: ' + args[1], 'color: #ea4335; font-weight: bold;');
                if (args[2] && args[2].value) console.log('    value: ' + args[2].value + ' ' + (args[2].currency || ''));
                if (args[2] && args[2].send_to) console.log('    send_to: ' + args[2].send_to);
            }

            return origGtag.apply(this, arguments);
        };
        console.log('  gtag interceptor active');
    } else {
        console.log('  gtag not found');
    }

    // ─── 4. NETWORK INTERCEPTION (ad platforms only) ─────────
    console.log('\n%c NETWORK INTERCEPTION (ad & tracking only)', 'background: #dc2626; color: white; padding: 5px; border-radius: 3px;');

    // fetch()
    var origFetch = window.fetch;
    window.fetch = function() {
        var url = typeof arguments[0] === 'string' ? arguments[0] : (arguments[0] && arguments[0].url ? arguments[0].url : '');
        var opts = arguments[1] || {};
        var platform = matchPlatform(url);

        if (platform) {
            var body = null;
            if (opts.body) {
                try { body = typeof opts.body === 'string' ? opts.body : JSON.stringify(opts.body); } catch(e) {}
            }
            addAdRequest({ type: 'fetch', platform: platform, url: url, method: opts.method || 'GET', body: body });
        }

        return origFetch.apply(this, arguments);
    };

    // XMLHttpRequest
    var origXhrOpen = XMLHttpRequest.prototype.open;
    var origXhrSend = XMLHttpRequest.prototype.send;

    XMLHttpRequest.prototype.open = function(method, url) {
        this._captureMethod = method;
        this._captureUrl = url;
        this._capturePlatform = matchPlatform(url);
        return origXhrOpen.apply(this, arguments);
    };

    XMLHttpRequest.prototype.send = function(body) {
        if (this._capturePlatform) {
            var bodyStr = null;
            if (body) { try { bodyStr = typeof body === 'string' ? body : JSON.stringify(body); } catch(e) {} }
            addAdRequest({ type: 'xhr', platform: this._capturePlatform, url: this._captureUrl, method: this._captureMethod || 'GET', body: bodyStr });
        }
        return origXhrSend.apply(this, arguments);
    };

    // sendBeacon
    var origBeacon = navigator.sendBeacon;
    if (origBeacon) {
        navigator.sendBeacon = function(url, data) {
            var platform = matchPlatform(url);
            if (platform) {
                var body = null;
                if (data) { try { body = typeof data === 'string' ? data : JSON.stringify(data); } catch(e) {} }
                addAdRequest({ type: 'beacon', platform: platform, url: url, method: 'POST', body: body });
            }
            return origBeacon.apply(this, arguments);
        };
    }

    // Image pixel fires
    var imgObserver = new MutationObserver(function(mutations) {
        mutations.forEach(function(mutation) {
            mutation.addedNodes.forEach(function(node) {
                if (node.tagName === 'IMG' && node.src) {
                    var platform = matchPlatform(node.src);
                    if (platform) {
                        addAdRequest({ type: 'img-pixel', platform: platform, url: node.src, method: 'GET', body: null });
                    }
                }
            });
        });
    });
    imgObserver.observe(document.body, { childList: true, subtree: true });

    console.log('  All interceptors active (fetch, XHR, beacon, img pixels)');

    // ─── 5. ALREADY-LOADED AD RESOURCES ──────────────────────
    console.log('\n%c ALREADY-LOADED AD RESOURCES', 'background: #065f46; color: white; padding: 5px; border-radius: 3px;');

    var perfEntries = performance.getEntriesByType('resource');
    var adResources = {};
    var adTransferSize = 0;

    perfEntries.forEach(function(e) {
        var platform = matchPlatform(e.name);
        if (platform) {
            if (!adResources[platform]) adResources[platform] = { count: 0, size: 0 };
            adResources[platform].count++;
            adResources[platform].size += e.transferSize || 0;
            adTransferSize += e.transferSize || 0;
        }
    });

    Object.keys(adResources).forEach(function(p) {
        console.log('  ' + p + ': ' + adResources[p].count + ' resources, ' + (adResources[p].size / 1024).toFixed(1) + ' KB');
    });
    console.log('  Total ad/tracking overhead: ' + (adTransferSize / 1024).toFixed(1) + ' KB');

    console.log('\n%c Navigate the site to capture events. Run adReport() anytime.', 'color: #FF8C42; font-weight: bold;');

    // ─── REPORT FUNCTION ─────────────────────────────────────
    window.adReport = function() {
        console.log('\n%c AD & TRACKING REPORT', 'background: #0891b2; color: white; font-size: 14px; padding: 10px; border-radius: 5px;');
        console.log('%c' + '='.repeat(55), 'color: #0891b2;');

        // Network requests by platform
        var byPlatform = {};
        adRequests.forEach(function(r) {
            if (!byPlatform[r.platform]) byPlatform[r.platform] = [];
            byPlatform[r.platform].push(r);
        });

        console.log('\n  AD NETWORK REQUESTS: ' + adRequests.length);
        Object.keys(byPlatform).sort(function(a, b) { return byPlatform[b].length - byPlatform[a].length; }).forEach(function(p) {
            console.log('    ' + p + ': ' + byPlatform[p].length + ' requests');
            byPlatform[p].forEach(function(r) {
                if (r.eventInfo.event) console.log('      ' + r.eventInfo.event + (r.eventInfo.eventId ? ' [event_id]' : '') + ' (' + r.elapsed + ')');
            });
        });

        // fbq events
        console.log('\n  FBQ EVENTS: ' + fbqEvents.length);
        fbqEvents.forEach(function(e) {
            if (e.action === 'track' || e.action === 'trackCustom') {
                console.log('    ' + e.event + (e.hasEventId ? ' [CAPI]' : '') + ' (' + e.elapsed + ')');
            }
        });

        // gtag events
        console.log('\n  GTAG EVENTS: ' + gtagEvents.length);
        gtagEvents.forEach(function(e) {
            if (e.command === 'event') console.log('    ' + e.eventOrId + ' (' + e.elapsed + ')');
        });

        // DataLayer
        var ecomEvents = dataLayerTimeline.filter(function(e) { return e.type === 'ecommerce'; });
        var consentEvents = dataLayerTimeline.filter(function(e) { return e.type === 'consent'; });
        var customEvents = dataLayerTimeline.filter(function(e) { return e.type === 'custom'; });

        console.log('\n  DATALAYER: ' + dataLayerTimeline.length + ' entries');
        console.log('    E-commerce events: ' + ecomEvents.length);
        ecomEvents.forEach(function(e) {
            var status = e.ecommerce.missing.length ? 'MISSING: ' + e.ecommerce.missing.join(', ') : 'complete';
            console.log('      ' + e.event + ' — ' + status);
        });
        console.log('    Consent events: ' + consentEvents.length);
        console.log('    Custom events: ' + customEvents.length);
        customEvents.forEach(function(e) { console.log('      ' + e.event); });

        // Performance
        console.log('\n  AD/TRACKING OVERHEAD: ' + (adTransferSize / 1024).toFixed(1) + ' KB');

        return {
            adRequests: adRequests, byPlatform: byPlatform,
            fbqEvents: fbqEvents, gtagEvents: gtagEvents,
            dataLayer: dataLayerTimeline, adResources: adResources,
            adTransferSize: adTransferSize
        };
    };

    // ─── HTML EXPORT ─────────────────────────────────────────
    window.downloadAdReportHTML = function() {
        var report = window.adReport();
        var h = window.location.hostname || 'local';
        var dt = new Date().toISOString().slice(0, 10);
        var duration = ((Date.now() - startTime) / 1000).toFixed(0);

        var ecomEvents = dataLayerTimeline.filter(function(e) { return e.type === 'ecommerce'; });
        var customEvents = dataLayerTimeline.filter(function(e) { return e.type === 'custom'; });

        // Platform rows
        var platformRows = Object.keys(report.byPlatform).sort(function(a, b) { return report.byPlatform[b].length - report.byPlatform[a].length; }).map(function(p) {
            var reqs = report.byPlatform[p];
            var events = reqs.filter(function(r) { return r.eventInfo.event; }).map(function(r) { return r.eventInfo.event; });
            var evStr = events.length ? events.join(', ') : '-';
            return '' + p + '' + reqs.length + '' + evStr + '';
        }).join('');

        // fbq rows
        var fbqRows = fbqEvents.filter(function(e) { return e.action === 'track' || e.action === 'trackCustom'; }).map(function(e) {
            var capiFlag = e.hasEventId ? ' CAPI' : '';
            var val = e.params.value ? e.params.value + ' ' + (e.params.currency || '') : '-';
            return '' + e.event + capiFlag + '' + val + '' + e.elapsed + '';
        }).join('');

        // gtag rows
        var gtagRows = gtagEvents.filter(function(e) { return e.command === 'event'; }).map(function(e) {
            var val = e.params.value ? e.params.value + ' ' + (e.params.currency || '') : '-';
            var sendTo = e.params.send_to || '-';
            return '' + e.eventOrId + '' + val + '' + sendTo + '';
        }).join('');

        // ecom rows
        var ecomRows = ecomEvents.map(function(e) {
            var statusColor = e.ecommerce.missing.length ? '#ef4444' : '#10b981';
            var status = e.ecommerce.missing.length ? 'Missing: ' + e.ecommerce.missing.join(', ') : 'Complete';
            return '' + e.event + '' + (e.ecommerce.itemCount || '-') + '' + e.ecommerce.totalValue + '' + status + '';
        }).join('');

        // Resource rows
        var resRows = Object.keys(adResources).sort(function(a, b) { return adResources[b].size - adResources[a].size; }).map(function(p) {
            return '' + p + '' + adResources[p].count + '' + (adResources[p].size / 1024).toFixed(1) + ' KB';
        }).join('');

        var html = 'Ad & Tracking Report - ' + h + '';
        html += '';

        html += '
Ad & Tracking Network Report
';
        html += '
Site: ' + h + ' | Date: ' + dt + ' | Capture duration: ' + duration + 's
';

        // KPIs
        html += '
';
        html += '
' + adRequests.length + '
Ad Requests
