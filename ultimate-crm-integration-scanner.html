        }
    });
    
    if (customEndpoints.length > 0) {
        scanner.apis = customEndpoints;
        console.log('✅ APIs/Endpoints custom detectados:', customEndpoints);
        scanner.totalIntegrations += customEndpoints.length;
    }
    
    // 9. DETECTAR SCRIPTS DE TERCEROS ÚNICOS
    console.log('\n%c🌐 ESCANEANDO SCRIPTS DE TERCEROS', 'background: #4338ca; color: white; padding: 5px 10px; border-radius: 3px; font-weight: bold;');
    
    const thirdPartyScripts = new Set();
    const scriptsWithSrc = document.querySelectorAll('script[src]');
    
    scriptsWithSrc.forEach(script => {
        try {
            const url = new URL(script.src);
            if (!url.hostname.includes(window.location.hostname)) {
                thirdPartyScripts.add(url.hostname);
            }
        } catch (e) {}
    });
    
    // Clasificar scripts no identificados
    const unidentifiedScripts = [];
    thirdPartyScripts.forEach(host => {
        // Lista de hosts conocidos (simplificada para brevedad)
        const knownHosts = [
            'google', 'facebook', 'stripe', 'paypal', 'klaviyo', 'hubspot',
            'zendesk', 'intercom', 'drift', 'typeform', 'segment', 'mixpanel'
        ];
        
        const isKnown = knownHosts.some(known => host.includes(known));
        if (!isKnown) {
            unidentifiedScripts.push(host);
        }
    });
    
    if (unidentifiedScripts.length > 0) {
        scanner.custom.unidentifiedScripts = unidentifiedScripts;
        console.log('⚠️ Scripts no identificados:', unidentifiedScripts);
    }
    
    // 10. DETECTAR IFRAMES Y EMBEDS
    console.log('\n%c🖼️ ESCANEANDO IFRAMES & EMBEDS', 'background: #ea580c; color: white; padding: 5px 10px; border-radius: 3px; font-weight: bold;');
    
    const iframes = document.querySelectorAll('iframe[src]');
    const embeds = [];
    
    iframes.forEach(iframe => {
        try {
            const url = new URL(iframe.src);
            if (!url.hostname.includes(window.location.hostname)) {
                embeds.push({
                    host: url.hostname,
                    type: identifyEmbedType(url.hostname),
                    width: iframe.width,
                    height: iframe.height
                });
            }
        } catch (e) {}
    });
    
    function identifyEmbedType(hostname) {
        if (hostname.includes('youtube')) return 'YouTube';
        if (hostname.includes('vimeo')) return 'Vimeo';
        if (hostname.includes('calendly')) return 'Calendly';
        if (hostname.includes('typeform')) return 'Typeform';
        if (hostname.includes('google') && hostname.includes('maps')) return 'Google Maps';
        if (hostname.includes('google') && hostname.includes('forms')) return 'Google Forms';
        if (hostname.includes('spotify')) return 'Spotify';
        if (hostname.includes('soundcloud')) return 'SoundCloud';
        return 'Unknown';
    }
    
    if (embeds.length > 0) {
        scanner.embeds = embeds;
        console.log('✅ Embeds detectados:', embeds);
    }
    
    // 11. RESUMEN FINAL Y REPORTE
    console.log('\n%c📋 RESUMEN COMPLETO DEL ESCANEO', 'background: linear-gradient(90deg, #10b981 0%, #059669 100%); color: white; font-size: 14px; padding: 10px 20px; border-radius: 5px; font-weight: bold;');
    console.log('%c══════════════════════════════════════════', 'color: #10b981;');
    
    // Crear tabla de resumen
    const summaryTable = {
        'Total Integraciones': scanner.totalIntegrations,
        'CRMs': Object.keys(scanner.crms).length,
        'Email Marketing': Object.keys(scanner.emailMarketing).length,
        'Chat/Support': Object.keys(scanner.chatSupport).length,
        'Automation': Object.keys(scanner.automation).length,
        'Payment': Object.keys(scanner.payments).length,
        'Forms': Object.keys(scanner.forms).length,
        'Analytics': Object.keys(scanner.analytics).length,
        'APIs Custom': scanner.apis.length,
        'Embeds': scanner.embeds.length
    };
    
    console.table(summaryTable);
    
    // Mostrar detalles por categoría
    if (Object.keys(scanner.crms).length > 0) {
        console.log('\n🎯 CRMs Detectados:');
        Object.values(scanner.crms).forEach(crm => {
            console.log(`  • ${crm.name}:`, crm.details);
        });
    }
    
    if (Object.keys(scanner.emailMarketing).length > 0) {
        console.log('\n📧 Email Marketing:');
        Object.values(scanner.emailMarketing).forEach(tool => {
            console.log(`  • ${tool.name}:`, tool.details);
        });
    }
    
    if (Object.keys(scanner.chatSupport).length > 0) {
        console.log('\n💬 Chat/Support:');
        Object.values(scanner.chatSupport).forEach(tool => {
            console.log(`  • ${tool.name}:`, tool.details);
        });
    }
    
    if (Object.keys(scanner.automation).length > 0) {
        console.log('\n🔧 Automation:');
        Object.values(scanner.automation).forEach(tool => {
            console.log(`  • ${tool.name}:`, tool.details);
        });
    }
    
    if (Object.keys(scanner.payments).length > 0) {
        console.log('\n💳 Payment Processors:');
        Object.values(scanner.payments).forEach(tool => {
            console.log(`  • ${tool.name}:`, tool.details);
        });
    }
    
    // Guardar resultados globalmente
    window.integrationScan = scanner;
    
    console.log('\n✅ Escaneo completo. Resultados guardados en: window.integrationScan');
    console.log('💡 Ejecuta downloadScanResults() para descargar el reporte completo');
    
    // Función para descargar resultados
    window.downloadScanResults = function() {
        const dataStr = JSON.stringify(scanner, null, 2);
        const dataUri = 'data:application/json;charset=utf-8,'+ encodeURIComponent(dataStr);
        const exportFileDefaultName = `integration_scan_${window.location.hostname}_${new Date().toISOString().slice(0,10)}.json`;
        
        const linkElement = document.createElement('a');
        linkElement.setAttribute('href', dataUri);
        linkElement.setAttribute('download', exportFileDefaultName);
        linkElement.click();
        console.log('✅ Reporte descargado:', exportFileDefaultName);
    };
    
    // Función para análisis profundo de una integración específica
    window.deepAnalyze = function(integration) {
        console.log('\n%c🔬 ANÁLISIS PROFUNDO: ' + integration.toUpperCase(), 'background: #dc2626; color: white; padding: 5px; border-radius: 3px;');
        
        // Buscar todos los elementos relacionados
        const related = document.querySelectorAll(`[class*="${integration}"], [id*="${integration}"], [data-*="${integration}"]`);
        console.log(`Elementos relacionados encontrados: ${related.length}`);
        
        // Buscar en window object
        const windowProps = Object.keys(window).filter(key => 
            key.toLowerCase().includes(integration.toLowerCase())
        );
        if (windowProps.length > 0) {
            console.log('Propiedades en window:', windowProps);
            windowProps.forEach(prop => {
                console.log(`  window.${prop}:`, window[prop]);
            });
        }
        
        // Buscar en localStorage
        const localStorageKeys = Object.keys(localStorage).filter(key =>
            key.toLowerCase().includes(integration.toLowerCase())
        );
        if (localStorageKeys.length > 0) {
            console.log('Keys en localStorage:', localStorageKeys);
        }
        
        // Buscar en cookies
        const cookies = document.cookie.split(';').filter(cookie =>
            cookie.toLowerCase().includes(integration.toLowerCase())
        );
        if (cookies.length > 0) {
            console.log('Cookies relacionadas:', cookies);
        }
    };
    
    console.log('\n💡 TIP: Ejecuta deepAnalyze("nombreIntegracion") para un análisis profundo de cualquier integración');
    
})();
