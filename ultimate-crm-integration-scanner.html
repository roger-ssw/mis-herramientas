        scanner.totalIntegrations++;
    }
    
    // ActiveCampaign
    if (window.ActiveCampaign || document.querySelector('script[src*="activehosted.com"]')) {
        scanner.emailMarketing.activeCampaign = true;
        console.log('✅ ActiveCampaign detectado');
        scanner.totalIntegrations++;
    }
    
    // 3. CHAT & SUPPORT
    console.log('\n%c💬 ESCANEANDO CHAT & SUPPORT', 'background: #dc2626; color: white; padding: 5px; border-radius: 3px;');
    
    // Intercom
    if (window.Intercom || window.intercomSettings) {
        scanner.chatSupport.intercom = true;
        console.log('✅ Intercom detectado');
        scanner.totalIntegrations++;

    // Drift
    if (window.drift || document.querySelector('script[src*="drift"]')) {
        scanner.chatSupport.drift = true;

        console.log('✅ Segment detectado');
        scanner.totalIntegrations++;

        console.log('✅ Hotjar detectado');
        scanner.totalIntegrations++;

    console.log('\n%c📋 RESUMEN DEL ESCANEO', 'background: linear-gradient(90deg, #10b981 0%, #059669 100%); color: white; font-size: 14px; padding: 10px; border-radius: 5px;');

    console.log('Total integraciones detectadas:', scanner.totalIntegrations);
    
    // Guardar resultados
    window.integrationScan = scanner;
    console.log('\n✅ Resultados disponibles en: window.integrationScan');
    
    // Función para descargar
    window.downloadScanResults = function() {
        const dataStr = JSON.stringify(scanner, null, 2);
        const dataUri = 'data:application/json;charset=utf-8,'+ encodeURIComponent(dataStr);
        const exportName = `scan_${window.location.hostname}_${new Date().toISOString().slice(0,10)}.json`;
        
        const link = document.createElement('a');
        link.setAttribute('href', dataUri);
        link.setAttribute('download', exportName);
        link.click();
        console.log('✅ Descargado:', exportName);
    };
